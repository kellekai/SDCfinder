45 int read_temperature()
46 {
47  int fd, fd_err, core = 0, ierr;
48  uint64_t temp;
49  char msr_file[32];
50  memset(msr_file, 0, 32);
51 
52 #ifdef SYS_getcpu
53  if (syscall(SYS_getcpu, &core, NULL, NULL) < 0) {
54  core = 0;
55  }
56 #endif
57 
58  fd_err = open(ErrFile, O_WRONLY|O_CREAT|O_APPEND, 0666);
59 
60  snprintf(msr_file, 31, "/dev/cpu/%u/msr", core);
61  fd = open(msr_file, O_RDONLY);
62  if (fd == -1) {
63  ierr = errno;
64  dprintf(fd_err, "could not OPEN msr file: %s\n", strerror(ierr));
65  ierr = 0;
66  return -1;
67  }
68 
69  ierr = pread(fd, &temp, sizeof(temp), TEMP_FIELD_OFFSET);
70  if (ierr == -1) {
71  ierr = errno;
72  dprintf(fd_err, "could not READ msr file: %s\n", strerror(ierr));
73  ierr = 0;
74  return -1;
75  }
76 
77  temp &= TEMP_FIELD_MASK;
78  temp >>= TEMP_FIELD_LOW_BIT;
79  temp = TCC_ACT_TEMP - temp;
80 
81  close(fd_err);
82  close(fd);
83 
84  return temp;
85 }
